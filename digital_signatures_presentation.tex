\section{Introduction}
% =====================================================
\begin{frame}{Introduction to Digital Signatures}
  \begin{itemize}
    \item A \textbf{digital signature} is a cryptographic mechanism used to verify the \textbf{authenticity} and \textbf{integrity} of a digital message or document.
    \item It is the electronic analogue of a handwritten signature, but far more secure.
    \item Based on \textbf{public-key (asymmetric) cryptography}: a \emph{private key} to sign, a \emph{public key} to verify.
    \item Provides three key security properties:
    \begin{itemize}
      \item \textbf{Authentication} -- confirms the identity of the sender.
      \item \textbf{Integrity} -- ensures the message was not altered.
      \item \textbf{Non-repudiation} -- sender cannot deny having signed the message.
    \end{itemize}
  \end{itemize}
\end{frame}

\section{Workflow}
% =====================================================
\begin{frame}{Workflow: How Digital Signatures Work}
  \textbf{1. Key Generation}
  \begin{itemize}
    \item Each user generates a pair of mathematically linked keys: a private key (kept secret) and a public key (shared openly).
  \end{itemize}
  \textbf{2. Signing (by Sender)}
  \begin{itemize}
    \item Compute a hash (message digest) of the document.
    \item Encrypt the hash using the sender's \emph{private key} $\rightarrow$ this is the digital signature.
    \item Send the original message along with the signature.
  \end{itemize}
  \textbf{3. Verification (by Receiver)}
  \begin{itemize}
    \item Recompute the hash of the received message.
    \item Decrypt the received signature using the sender's \emph{public key}.
    \item If both hash values match $\Rightarrow$ signature is \textbf{valid}; otherwise \textbf{invalid / tampered}.
  \end{itemize}
\end{frame}

\section{RSA Digital Signature}
% =====================================================
\begin{frame}{RSA Digital Signature Scheme}
  \textbf{Key Generation}
  \begin{enumerate}
    \item Choose two large primes $p, q$. Compute $n = pq$.
    \item Compute $\varphi(n) = (p-1)(q-1)$.
    \item Choose public exponent $e$ such that $1 < e < \varphi(n)$ and $\gcd(e,\varphi(n))=1$.
    \item Compute private exponent $d \equiv e^{-1} \pmod{\varphi(n)}$.
    \item Public key: $(n,e)$ \quad Private key: $(n,d)$.
  \end{enumerate}
  \textbf{Signing:} \quad $s \equiv m^{d} \pmod{n}$

  \textbf{Verification:} \quad accept if \; $m \equiv s^{e} \pmod{n}$
\end{frame}

\begin{frame}{RSA Digital Signature -- Small Example}
  \textbf{Key Generation}
  \begin{align*}
    p = 5, \quad q = 11 &\quad\Rightarrow\quad n = pq = 55 \\
    \varphi(n) &= (p-1)(q-1) = 4 \times 10 = 40 \\
    e &= 3 \quad (\gcd(3,40)=1) \\
    d &= e^{-1} \bmod 40 = 27 \quad (\text{since } 3 \times 27 = 81 \equiv 1 \bmod 40)
  \end{align*}
  Public key $(n,e) = (55,3)$, \quad Private key $(n,d) = (55,27)$

  \textbf{Signing message} $m = 2$:
  \[
    s \equiv m^{d} \pmod n \;=\; 2^{27} \bmod 55 \;=\; 18
  \]

  \textbf{Verification:}
  \[
    s^{e} \bmod n \;=\; 18^{3} \bmod 55 \;=\; 2 \;=\; m \quad \checkmark \; \text{Signature valid!}
  \]
\end{frame}

\section{ElGamal Digital Signature}
% =====================================================
\begin{frame}{ElGamal Digital Signature Scheme}
  \textbf{Key Generation}
  \begin{enumerate}
    \item Choose a large prime $p$ and a generator $g$ of $\mathbb{Z}_p^{*}$.
    \item Choose private key $x$, $1 < x < p-1$.
    \item Compute public key $y \equiv g^{x} \pmod p$.
    \item Public key: $(p, g, y)$ \quad Private key: $x$.
  \end{enumerate}
  \textbf{Signing message $m$:}
  \begin{enumerate}
    \item Choose random $k$ with $\gcd(k, p-1) = 1$.
    \item Compute $r \equiv g^{k} \pmod p$.
    \item Compute $s \equiv k^{-1}(m - xr) \pmod{p-1}$.
    \item Signature: $(r,s)$.
  \end{enumerate}
  \textbf{Verification:} accept if \; $g^{m} \equiv y^{r}\, r^{s} \pmod p$
\end{frame}

\begin{frame}{ElGamal Digital Signature -- Small Example}
  \textbf{Key Generation}
  \begin{align*}
    p = 23, \quad g = 5, \quad x = 6 \;(\text{private})\\
    y \equiv g^{x} \bmod p = 5^{6} \bmod 23 = 8 \;(\text{public})
  \end{align*}
  Public key $(p,g,y) = (23,5,8)$, \quad Private key $x = 6$

  \textbf{Signing message} $m = 10$, choose $k = 3$ $(\gcd(3,22)=1)$:
  \begin{align*}
    r &\equiv g^{k} \bmod p = 5^{3} \bmod 23 = 10 \\
    s &\equiv k^{-1}(m - xr) \bmod (p-1) = 15 \times (10 - 60) \bmod 22 = 20
  \end{align*}
  Signature: $(r,s) = (10, 20)$

  \textbf{Verification:}
  \[
    g^{m} \bmod p = 5^{10} \bmod 23 = 9, \qquad
    y^{r} r^{s} \bmod p = 8^{10}\times 10^{20} \bmod 23 = 9
  \]
  Both sides equal $9$ \; $\checkmark$ \; Signature valid!
\end{frame}
